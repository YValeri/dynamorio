57 typedef struct _lookup_entry_t{
65  inline bool contains(app_pc pc, plc_lookup_type_t lookup) const{
66  if(lookup == PLC_LOOKUP_MODULE || (total && symbols.empty())){
67 #ifdef WINDOWS
68  return range.contains(pc);
69 #else
70  //Either we are looking for the whole module, or for a symbol in a total module without symbols as exceptions
71  if(contiguous){
72  return range.contains(pc);
73  }else{
74  //The module isn't contiguous, we need to check each segment
75  size_t segsize = segments.size();
76  for(size_t i = 0; i < segsize; i++){
77  if(segments[i].contains(pc)){
78  return true;
79  }
80  }
81 
82  return false;
83  }
84 #endif
85  }else{
86  //We are looking at a symbol
87  if(!range.contains(pc)){
88  //If the address isn't even in the full range of the module, it won't be found
89  return false;
90  }else{
91  //Note : checking for each range of segments would be possible, but performance-wise wouldn't be much better than looking for symbols
92  size_t symsize = symbols.size();
93  for(size_t i = 0; i < symsize; i++){
94  if(symbols[i].contains(pc)){
95  return !total;
96  }
97  } //If it was "total", then the symbols would be exceptions
98 
99  return total; //In total state with symbols, not finding the right symbol means it's not an exception
100  }
101  }
102  }
